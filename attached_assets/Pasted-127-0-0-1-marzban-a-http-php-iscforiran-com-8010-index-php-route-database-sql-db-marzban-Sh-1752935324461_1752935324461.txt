    a.username AS admin_username,
    COUNT(t.id) AS total_transactions,
    SUM(t.amount) AS total_invoice_amount
FROM 
    transactions t
JOIN 
    admins a ON t.admin_id = a.id
WHERE 
    t.event_timestamp BETWEEN '2025-07-08 00:00:00' AND '2025-07-18 23:59:59' -- <-- تاریخ را اینجا ویرایش کنید
GROUP BY 
    a.username
ORDER BY 
    a.username ASC;
